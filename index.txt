// DEĞİŞİKLİK 2: Kategoriye Göre Uygunsuzluk — akordiyonlu render
// ── Her kategorideki her madde için tüm denetimlerin toplam
//    Evet/Hayır/Eksik/Gerekli Değil sayıları hesaplanır
// ════════════════════════════════════════════════
globalThis.renderKatBar = function(v) {
  // Önce her kategori için toplam uygunsuz sayısını hesapla
  const kt = {};
 ['A','B','C','D','E','F','G','H','I'].forEach(k => { kt[k] = 0; });
  v.forEach(d => {
    const kat = d.kat || {};
    Object.keys(kt).forEach(k => { kt[k] += (kat[k] || 0); });
  });
  const ks  = Object.entries(kt).sort((a,b) => b[1]-a[1]);
  const km  = ks[0]?.[1] || 1;

  // Her madde için tüm denetimlerdeki toplam değerleri hesapla
  // Sheets'ten gelen detaylı madde verisi varsa kullan, yoksa kat toplamını böl
  // Sheets'ten madde bazlı veri gelmiyorsa kategori toplamını madde sayısına bölerek tahmin
  const maddeToplamlar = {}; // {no: {evet, hayir, eksik, ger}}
  // Tüm denetimlerin answers'ını topla — Sheets'ten gelen verilerde answers olmayabilir
  // Bu durumda sadece kat toplamı gösteririz
  v.forEach(d => {
    if (!d.answers) return;
    Object.entries(d.answers).forEach(([no, val]) => {
      const n = parseInt(no);
      if (!maddeToplamlar[n]) maddeToplamlar[n] = {evet:0,hayir:0,eksik:0,ger:0};
      const vl = (val||'').toLowerCase();
      if (vl==='evet/var')         maddeToplamlar[n].evet++;
      else if (vl==='hayır/yok')   maddeToplamlar[n].hayir++;
      else if (vl==='eksik')       maddeToplamlar[n].eksik++;
      else if (vl==='gerekli değil') maddeToplamlar[n].ger++;
    });
  });

  const hasMaddeData = Object.keys(maddeToplamlar).length > 0;

  const html = ks.map(([k, n]) => {
    const maddeArr = KAT_MADDELER[k] || [];
    const pct      = Math.round(n / km * 100);
    const grad     = KAT_RENKLER[k] || '#64748b';
    const label    = KAT_ADLAR[k] || k;
    const totalMadde = maddeArr.length;

    // Madde detay satırları
    let detailRows = '';
    if (hasMaddeData) {
      detailRows = maddeArr.map(no => {
        const m    = maddeToplamlar[no] || {evet:0,hayir:0,eksik:0,ger:0};
        const ad   = MADDE_ADLARI_DASH[no] || ('Madde ' + no);
        const uyg  = m.hayir + m.eksik;
        const adKisa = ad.length > 55 ? ad.substring(0,55)+'…' : ad;
        return '<div class="kat-madde-row">\n          <div><span class="kat-madde-no">' + (no) + '.</span><span class="kat-madde-ad">' + (adKisa) + '</span></div>\n          <div class="kat-madde-cell kat-cell-evet">' + (m.evet||'–') + '</div>\n          <div class="kat-madde-cell kat-cell-hayir">' + (m.hayir > 0 ? m.hayir : '–') + '</div>\n          <div class="kat-madde-cell kat-cell-eksik">' + (m.eksik > 0 ? m.eksik : '–') + '</div>\n          <div class="kat-madde-cell kat-cell-ger">' + (m.ger > 0 ? m.ger : '–') + '</div>\n        </div>';
      }).join('');

      // Footer toplamları
      const totEvet  = maddeArr.reduce((s,no) => s + (maddeToplamlar[no]?.evet||0), 0);
      const totHayir = maddeArr.reduce((s,no) => s + (maddeToplamlar[no]?.hayir||0), 0);
      const totEksik = maddeArr.reduce((s,no) => s + (maddeToplamlar[no]?.eksik||0), 0);
      const totGer   = maddeArr.reduce((s,no) => s + (maddeToplamlar[no]?.ger||0), 0);

      detailRows += '<div class="kat-detail-footer">\n        <div class="kat-footer-lbl">TOPLAM (' + (totalMadde) + ' madde)</div>\n        <div class="kat-footer-cell kat-cell-evet">' + (totEvet) + '</div>\n        <div class="kat-footer-cell kat-cell-hayir">' + (totHayir) + '</div>\n        <div class="kat-footer-cell kat-cell-eksik">' + (totEksik) + '</div>\n        <div class="kat-footer-cell kat-cell-ger">' + (totGer) + '</div>\n      </div>';
    } else {
      // Sheets'ten sadece kat toplamı geldiyse özet göster
      detailRows = '<div style="padding:10px 14px;font-size:11px;color:var(--mut2)">\n        Bu kategori için ' + (totalMadde) + ' madde · Toplam uygunsuzluk: <strong style="color:var(--red)">' + (n) + '</strong><br>\n        <span style="font-size:10px;color:var(--mut)">Madde bazlı detay için form üzerinden denetim girin</span>\n      </div>';
    }

    return '<div class="kat-accordion">\n      <div class="kat-bar-row" onclick="katToggle(this)">\n        <div class="kat-bar-lbl">' + (label) + '</div>\n        <div class="bar-trk">\n          <div class="bar-fill" style="width:' + (pct) + '%;background:' + (grad) + '">' + (n) + '</div>\n        </div>\n        <div class="kat-toggle">▼</div>\n      </div>\n      <div class="kat-detail">\n        <div class="kat-detail-header">\n          <div>Madde</div>\n          <div style="text-align:center">✅ Evet</div>\n          <div style="text-align:center">❌ Hayır</div>\n          <div style="text-align:center">⚠️ Eksik</div>\n          <div style="text-align:center">— G.Değil</div>\n        </div>\n        ' + (detailRows) + '\n      </div>\n    </div>';
  }).join('');

  document.getElementById('katBar').innerHTML = html || '<span style="color:var(--mut);font-size:12px">Veri yok – Denetim ekleyin</span>';
}

// DEĞİŞİKLİK 3: Risk Sıralaması — % bazlı hesaplama
// Formül: (hayır+eksik) / 138 * 100
// İşaretlenmeyen maddeler "Evet" kabul edilir (0 uygunsuz)
// ════════════════════════════════════════════════
let riskTumVeri = [];
let riskSayfa = 0;
const RISK_SAYFA_BOYUT = 15;

globalThis.renderRiskList = function(v) {
  riskTumVeri = [...v].map(d => {
    const uyg = d.hayir + d.eksik;
    const pct = parseFloat((uyg / 138 * 100).toFixed(1));
    return { ...d, uyg, pct };
  }).sort((a, b) => b.pct - a.pct);
  riskSayfa = 0;
  const aramaEl = document.getElementById('riskArama');
  if (aramaEl) aramaEl.value = '';
  _riskRender();
}

globalThis.riskFiltrele = function() {
  riskSayfa = 0;
  _riskRender();
}

globalThis._riskRender = function() {
  const arama = (document.getElementById('riskArama')?.value || '').toLowerCase().trim();
  const filtreVeri = arama
    ? riskTumVeri.filter(d => (d.okul || '').toLowerCase().includes(arama))
    : riskTumVeri;
  const toplamSayfa = Math.ceil(filtreVeri.length / RISK_SAYFA_BOYUT);
  if (riskSayfa >= toplamSayfa) riskSayfa = Math.max(0, toplamSayfa - 1);
  const bas = riskSayfa * RISK_SAYFA_BOYUT;
  const gosterilecek = filtreVeri.slice(bas, bas + RISK_SAYFA_BOYUT);
  const sayacEl = document.getElementById('riskSayac');
  if (sayacEl) sayacEl.textContent = filtreVeri.length > 0
    ? (bas+1) + '–' + (Math.min(bas+RISK_SAYFA_BOYUT, filtreVeri.length)) + ' / ' + (filtreVeri.length) + ' okul'
    : '';
  document.getElementById('riskList').innerHTML = gosterilecek.map((d, i) => {
    const sira = bas + i + 1;
    const c = d.pct >= 30 ? '#f87171' : d.pct >= 15 ? '#fbbf24' : '#34d399';
    const ad = d.okul.length > 32 ? d.okul.substring(0,32)+'…' : d.okul;
    return '<div class="risk-row">\n      <div style="width:22px;text-align:right;font-size:10px;color:var(--mut);flex-shrink:0">' + (sira) + '.</div>\n      <div class="risk-okul">' + (ad) + '<span>' + (d.tur||'–') + ' · ' + (d.tarih||'–') + '</span></div>\n      <div class="risk-bar"><div class="risk-bar-f" style="width:' + (Math.min(100,d.pct)) + '%;background:' + (c) + '"></div></div>\n      <div class="risk-score" style="color:' + (c) + '">%' + (d.pct.toFixed(1)) + '</div>\n    </div>';
  }).join('') || '<span style="color:var(--mut);font-size:12px">Veri yok</span>';
  _riskSayfaButonlari(toplamSayfa);
}

globalThis._riskSayfaButonlari = function(toplamSayfa) {
  const el = document.getElementById('riskSayfalama');
  if (!el) return;
  if (toplamSayfa <= 1) { el.innerHTML = ''; return; }
  const stil = (a) => 'class="risk-pg-btn' + (a?' aktif':'') + '"';
  let html = '<button ' + (stil(false)) + ' onclick="riskSayfaDegis(-1)">←</button>';
  for (let i = 0; i < toplamSayfa; i++) {
    if (toplamSayfa > 7 && Math.abs(i-riskSayfa) > 2 && i !== 0 && i !== toplamSayfa-1) {
      if (i===1||i===toplamSayfa-2) html += '<span style="color:var(--mut);padding:0 3px">…</span>';
      continue;
    }
    html += '<button ' + (stil(i===riskSayfa)) + ' onclick="riskSayfaDegis(' + (i) + ',true)">' + (i+1) + '</button>';
  }
  html += '<button ' + (stil(false)) + ' onclick="riskSayfaDegis(1)">→</button>';
  el.innerHTML = html;
}

globalThis.riskSayfaDegis = function(val, mutlak=false) {
  const arama = (document.getElementById('riskArama')?.value||'').toLowerCase().trim();
  const filtre = arama ? riskTumVeri.filter(d=>(d.okul||'').toLowerCase().includes(arama)) : riskTumVeri;
  const toplamSayfa = Math.ceil(filtre.length / RISK_SAYFA_BOYUT);
  riskSayfa = mutlak ? val : Math.max(0, Math.min(toplamSayfa-1, riskSayfa+val));
  _riskRender();
}

globalThis.animNum = function(id, t) {
  const el = document.getElementById(id);
  if (!el) return;
  const s   = parseInt(el.textContent) || 0;
  const dur = 700, start = performance.now();
  function frame(now) {
    const p = Math.min((now - start) / dur, 1);
    const e = 1 - Math.pow(1-p, 3);
    el.textContent = Math.round(s + (t-s) * e);
    if (p < 1) requestAnimationFrame(frame);
  }
  requestAnimationFrame(frame);
}

// JSONP yardımcı — file:// protokolünden CORS olmadan çalışır
function _jsonpFetch(url) {
  return new Promise(function(resolve, reject) {
    var cbName = '__isgCb_' + Date.now();
    var script = document.createElement('script');
    var done = false;
    var timer = setTimeout(function() {
      if (!done) { done = true; delete window[cbName]; script.remove();
        reject(new Error('Zaman aşımı — Apps Script URL ve dağıtım ayarlarını kontrol edin')); }
    }, 12000);
    window[cbName] = function(data) {
      if (!done) { done = true; clearTimeout(timer); delete window[cbName]; script.remove(); resolve(data); }
    };
    script.onerror = function() {
      if (!done) { done = true; clearTimeout(timer); delete window[cbName]; script.remove();
        reject(new Error('Bağlantı hatası — URL doğru mu ve dağıtım "Erişim: Herkes" mi?')); }
    };
    script.src = url + '&callback=' + cbName;
    document.head.appendChild(script);
  });
}

// JSONP yardımcı — Apps Script callback parametresini destekler
globalThis.jsonpFetch = function(url) {
  return new Promise((resolve, reject) => {
    const cbName = '__isgCb_' + Date.now();
    const script = document.createElement('script');
    let done = false;

    window[cbName] = function(data) {
      done = true;
      delete window[cbName];
      script.remove();
      resolve(data);
    };

    script.src     = url + '&callback=' + cbName;
    script.onerror = () => {
      if (!done) {
        delete window[cbName];
        script.remove();
        reject(new Error('JSONP yüklenemedi. Apps Script "Erişim: Herkes" olarak dağıtılmalı.'));
      }
    };
    // 8 saniye timeout
    setTimeout(() => {
      if (!done) {
        delete window[cbName];
        script.remove();
        reject(new Error('Zaman aşımı (8s). Apps Script URL\'nizi ve dağıtım ayarlarını kontrol edin.'));
      }
    }, 8000);

    document.head.appendChild(script);
  });
}

globalThis.parseOkulDosyasi = function(file) {
  const isXlsx = file.name.endsWith('.xlsx') || file.name.endsWith('.xls');
  const reader  = new FileReader();
  if (isXlsx) {
    if (!window.XLSX) {
      const script = document.createElement('script');
      script.src   = 'https://cdnjs.cloudflare.com/ajax/libs/xlsx/0.18.5/xlsx.full.min.js';
      script.onload = () => reader.readAsArrayBuffer(file);
      document.head.appendChild(script);
    } else { reader.readAsArrayBuffer(file); }
    reader.onload = e => {
      try {
        const wb   = XLSX.read(e.target.result, {type:'array'});
        const ws   = wb.Sheets[wb.SheetNames[0]];
        const rows = XLSX.utils.sheet_to_json(ws, {header:1, defval:''});
        islemeOkulRows(rows);
      } catch(err) { toast('Excel okuma hatası: ' + err.message, 't-err'); }
    };
  } else {
    reader.readAsText(file, 'utf-8');
    reader.onload = e => {
      const rows = e.target.result.split('\n').map(r =>
        r.split(/[,;|\t]/).map(c => c.trim().replace(/^"|"$/g,''))
      );
      islemeOkulRows(rows);
    };
  }
}

globalThis.islemeOkulRows = function(rows) {
  if (rows.length < 2) { toast('Dosyada veri bulunamadı.', 't-err'); return; }
  const header = rows[0].map(h => String(h).toLowerCase().trim());
  const colMap = {
    ad:      findCol(header, ['okul adı','okul','ad','kurum adı']),
    tur:     findCol(header, ['okul türü','tür','tur','tip']),
    mudur:   findCol(header, ['müdür','mudur','müdür adı']),
    tel:     findCol(header, ['telefon','tel','gsm']),
    adres:   findCol(header, ['adres','address']),
    tehlike: findCol(header, ['tehlike sınıfı','tehlike','sinif'])
  };
  okulListesi = [];
  for (let i = 1; i < rows.length; i++) {
    const r = rows[i];
    if (!r || r.every(c => !c)) continue;
    const okul = {
      ad:      colMap.ad      >= 0 ? String(r[colMap.ad]     ||'').trim() : String(r[0]||'').trim(),
      tur:     colMap.tur     >= 0 ? String(r[colMap.tur]    ||'').trim() : '',
      mudur:   colMap.mudur   >= 0 ? String(r[colMap.mudur]  ||'').trim() : '',
      tel:     colMap.tel     >= 0 ? String(r[colMap.tel]    ||'').trim() : '',
      adres:   colMap.adres   >= 0 ? String(r[colMap.adres]  ||'').trim() : '',
      tehlike: colMap.tehlike >= 0 ? String(r[colMap.tehlike]||'').trim() : ''
    };
    if (okul.ad) okulListesi.push(okul);
  }
  if (okulListesi.length === 0) { toast('Okul listesi boş veya sütun başlıkları tanınmadı.', 't-err'); return; }
  okulSelectiDoldur(okulListesi);
  toast('✅ ' + okulListesi.length + ' okul yüklendi', 't-ok');
}

globalThis.findCol = function(header, variants) {
  for (const v of variants) {
    const i = header.findIndex(h => h.includes(v));
    if (i >= 0) return i;
  }
  return -1;
}

// ── Fotoğraf sıkıştırma — POST limitini aşmamak için ──
function _compressFoto(file, maxW, quality, cb) {
  var reader = new FileReader();
  reader.onload = function(ev) {
    var img = new Image();
    img.onload = function() {
      var w   = Math.min(img.width, maxW);
      var h   = Math.round(img.height * w / img.width);
      var cnv = document.createElement('canvas');
      cnv.width = w; cnv.height = h;
      cnv.getContext('2d').drawImage(img, 0, 0, w, h);
      cb(cnv.toDataURL('image/jpeg', quality));
    };
    img.src = ev.target.result;
  };
  reader.readAsDataURL(file);
}

globalThis.fotoYukle = function(no, inp) {
  if (!photos[no]) photos[no] = [];
  const kalan    = 5 - photos[no].length;
  const dosyalar = Array.from(inp.files).slice(0, Math.max(0, kalan));
  if (dosyalar.length === 0) { toast('Bu madde için max 5 fotoğraf eklenebilir', 't-warn'); return; }
  var yuklenen = 0;
  dosyalar.forEach(function(dosya) {
    _compressFoto(dosya, 1200, 0.75, function(compressed) {
      photos[no].push(compressed);
      yuklenen++;
      if (yuklenen === dosyalar.length) _fotoPrevGuncelle(no);
    });
  });
  inp.value = '';
}

globalThis._fotoPrevGuncelle = function(no) {
  const prev    = document.getElementById('fotoPrevMini_' + no);
  if (!prev) return;
  const fotoArr = photos[no] || [];
  prev.innerHTML = fotoArr.map((src, i) =>
    '<div class="foto-thumb-mini" title="Sil">\n      <img src="' + (src) + '" onclick="fotoSil(' + (no) + ',' + (i) + ')" title="Sil (tıkla)">\n      <span class="foto-sil-btn" onclick="fotoSil(' + (no) + ',' + (i) + ')">✕</span>\n    </div>').join('');
  const btn = prev.closest('.foto-alan')?.querySelector('.foto-ekle-btn');
  if (btn) btn.textContent = fotoArr.length > 0 ? '📷 Fotoğraf Ekle (' + (fotoArr.length) + '/5)' : '📷 Fotoğraf Ekle';
}

globalThis.fotoSil = function(no, idx) {
  if (photos[no]) {
    photos[no].splice(idx, 1);
    if (photos[no].length === 0) delete photos[no];
    _fotoPrevGuncelle(no);
  }
}

globalThis.updateUI = function() {
  for (let i = 0; i < TOTAL_STEPS; i++) {
    const c = document.getElementById('step' + i);
    if (c) c.classList.toggle('active', i === currentStep);
  }
  for (let i = 0; i < TOTAL_STEPS; i++) {
    const p = document.getElementById('pill' + i);
    if (!p) continue;
    p.classList.remove('active','done');
    if (i === currentStep) p.classList.add('active');
    else if (i < currentStep) p.classList.add('done');
  }
  const pct = Math.round(((currentStep + 1) / TOTAL_STEPS) * 100);
  document.getElementById('progFill').style.width = pct + '%';
  document.getElementById('progLbl').textContent  = 'Adım ' + (currentStep+1) + ' / ' + TOTAL_STEPS;
  document.getElementById('progPct').textContent  = pct + '%';
  document.getElementById('btnPrev').style.display   = currentStep > 0 ? 'block' : 'none';

  document.getElementById('btnNext').style.display   = last ? 'none'  : 'block';
  document.getElementById('btnSubmit').style.display = last ? 'block' : 'none';

globalThis.okulTuruBulClient = function(ad) {
  const a = (ad || '').toUpperCase();
  if (a.includes('ORTAOKUL'))                                                           return 'Ortaokul';
  if (a.includes('İLKOKUL') || a.includes('ILKOKUL'))                                  return 'İlkokul';
  if (a.includes('LİSE') || a.includes('LISE') || a.includes('ANADOLU') ||
      a.includes('FEN L') || a.includes('MESLEK') || a.includes('PROGRAMLI') ||
      a.includes('IMAM')  || a.includes('İMAM'))                                        return 'Lise';
  if (a.includes('ANAOKUL'))                                                            return 'Anaokul';
  return 'Diğer';
}

globalThis.planAcListesi = [];  // Sheets'ten yüklenen liste (plan formu için)

globalThis.planOkulListesiYukle = async function() {
  if (!APPS_SCRIPT_URL) { toast('Önce Apps Script URL\'sini kaydedin', 't-warn'); return; }
  const btn  = document.getElementById('planSheetsBtn');
  const spin = document.getElementById('planSheetsSpin');
  btn.disabled = true; spin.classList.add('show');
  const buildUrl = (action) =>
    APPS_SCRIPT_URL + (APPS_SCRIPT_URL.includes('?') ? '&' : '?') + 'action=' + action + '&t=' + Date.now();
  try {
    let data = null;
    try {
      const resp = await fetch(buildUrl('getOkulListesi'), { method:'GET', redirect:'follow' });
      if (resp.ok) {
        try { data = JSON.parse(await resp.text()); } catch(pe) { data = await jsonpFetch(buildUrl('getOkulListesi')); }
      } else { data = await jsonpFetch(buildUrl('getOkulListesi')); }
    } catch(e) { data = await jsonpFetch(buildUrl('getOkulListesi')); }
    if (data && data.status === 'OK' && Array.isArray(data.okullar)) {
      planAcListesi = data.okullar;
      btn.textContent = '✅ ' + planAcListesi.length + ' okul yüklendi';
      toast('✅ ' + planAcListesi.length + ' okul yüklendi', 't-ok');
    } else {
      // Sheets'ten gelmezse DENETIMLER'den benzersiz okul listesi oluştur
      planAcListesi = [...new Set(DENETIMLER.map(d => d.okul))].filter(Boolean).map(ad => ({ad}));
      toast('Sheets\'ten yüklenemedi, mevcut denetimlerden liste oluşturuldu', 't-warn');
    }
  } catch(e) {
    toast('Hata: ' + e.message, 't-err');
  } finally { spin.classList.remove('show'); btn.disabled = false; }
}

// Mevcut okulListesi varsa planAcListesi'ne kopyala
globalThis.planAcListesiniGuncelle = function() {
  if (okulListesi && okulListesi.length > 0) {
    planAcListesi = okulListesi;
  }
}

globalThis.acFiltrele = function() {
  const inp = document.getElementById('pfOkul');
  const dd  = document.getElementById('acDropdown');
  if (!inp || !dd) return;
  const q = inp.value.toLowerCase().trim();
  // Kaynak: planAcListesi, yoksa okulListesi
  const kaynak = planAcListesi.length > 0 ? planAcListesi : okulListesi;
  if (kaynak.length === 0) {
    dd.innerHTML = '<div class="ac-empty">📊 Önce "Sheets\'ten Yükle" butonuna tıklayın</div>';
    dd.classList.add('open'); return;
  }
  const sonuclar = q.length === 0
    ? kaynak.slice(0, 50)
    : kaynak.filter(o => (o.ad||'').toLowerCase().includes(q)).slice(0, 50);
  if (sonuclar.length === 0) {
    dd.innerHTML = '<div class="ac-empty">Sonuç bulunamadı — yine de yazabilirsiniz</div>';
    dd.classList.add('open'); return;
  }
  dd.innerHTML = sonuclar.map(o => {
    const sub = [o.tur, o.mudur].filter(Boolean).join(' · ');
    return '<div class="ac-item" onclick="acSec(this)"'
      + ' data-ad="' + (o.ad||'') + '"'
      + ' data-mudur="' + (o.mudur||'') + '"'
      + ' data-tel="' + (o.tel||'') + '"'
      + ' data-adres="' + (o.adres||'') + '"'
      + ' data-tur="' + (o.tur||'') + '">'
      + '<div>' + (o.ad||'') + '</div>'
      + (sub ? '<div class="ac-item-sub">' + sub + '</div>' : '')
      + '</div>';
  }).join('');
  dd.classList.add('open');
}

globalThis.acSec = function(el) {
  const ad    = el.dataset.ad    || '';
  const mudur = el.dataset.mudur || '';
  const tel   = el.dataset.tel   || '';
  const adres = el.dataset.adres || '';
  const tur   = el.dataset.tur   || '';
  const inp = document.getElementById('pfOkul');
  if (inp) inp.value = ad;
  const m = document.getElementById('pfMudur'); if (m && mudur) m.value = mudur;
  const t = document.getElementById('pfTel');   if (t && tel)   t.value = tel;
  const a = document.getElementById('pfAdres'); if (a && adres) a.value = adres;
  const s = document.getElementById('pfTur');   if (s && tur)   s.value = tur;
  document.getElementById('acDropdown').classList.remove('open');
}

globalThis.planOkulKaydet = async function() {
  const okul  = document.getElementById('pfOkul').value.trim();
  const mudur = document.getElementById('pfMudur').value.trim();
  const tel   = document.getElementById('pfTel').value.trim();
  const adres = document.getElementById('pfAdres').value.trim();
  const tarih = document.getElementById('pfTarih').value;
  const tur   = document.getElementById('pfTur').value;
  if (!okul) { toast('Okul adı zorunlu', 't-warn'); return; }
  const yeni = { id: Date.now(), okul, mudur, tel, adres, tarih, tur };
  // Önce local ekle (hızlı görünüm)
  PLAN_OKULLAR.push(yeni);
  try { localStorage.setItem('ISG_PLAN_OKULLAR', JSON.stringify(PLAN_OKULLAR)); } catch(e){}
  ['pfOkul','pfMudur','pfTel','pfAdres','pfTarih'].forEach(id => { const el=document.getElementById(id); if(el) el.value=''; });
  document.getElementById('pfTur').value = '';
  planOkulFormKapat();
  planListeRender();
  toast('💾 Kaydediliyor…', '');
  // Sheets'e gönder
  if (APPS_SCRIPT_URL) {
    try {
      const resp = await fetch(APPS_SCRIPT_URL, {
        method: 'POST', redirect: 'follow',
        headers: {'Content-Type': 'text/plain;charset=utf-8'},
        body: JSON.stringify({ action: 'planEkle', plan: yeni })
      });
      const txt = await resp.text();
      const data = JSON.parse(txt);
      if (data.status === 'OK') {
        toast('✅ Plana eklendi ve Sheets\'e kaydedildi', 't-ok');
      } else {
        toast('⚠️ Sheets kaydı başarısız: ' + (data.mesaj||''), 't-warn');
      }
    } catch(e) {
      toast('⚠️ Sheets\'e ulaşılamadı, yerel kaydedildi', 't-warn');
    }
  } else {
    toast('✅ Plana eklendi (Sheets URL girilmedi)', 't-ok');
  }
}
globalThis.planOkulSil = async function(id) {
  if (!confirm('Bu planı silmek istediğinize emin misiniz?')) return;
  PLAN_OKULLAR = PLAN_OKULLAR.filter(p => p.id !== id);
  try { localStorage.setItem('ISG_PLAN_OKULLAR', JSON.stringify(PLAN_OKULLAR)); } catch(e){}
  planListeRender();
  if (APPS_SCRIPT_URL) {
    try {
      await fetch(APPS_SCRIPT_URL, {
        method: 'POST', redirect: 'follow',
        headers: {'Content-Type': 'text/plain;charset=utf-8'},
        body: JSON.stringify({ action: 'planSil', id: id })
      });
    } catch(e) { console.warn('Sheets silme hatası:', e); }
  }
  toast('Plandan silindi', '');
}
globalThis.planSheetstenYukle = async function() {
  if (!APPS_SCRIPT_URL) return;
  const buildUrl = (action) =>
    APPS_SCRIPT_URL + (APPS_SCRIPT_URL.includes('?') ? '&' : '?') + 'action=' + action + '&t=' + Date.now();
  try {
    let data = null;
    try {
      const resp = await fetch(buildUrl('getPlanlar'), { method:'GET', redirect:'follow' });
      if (resp.ok) {
        try { data = JSON.parse(await resp.text()); } catch(pe) { data = await jsonpFetch(buildUrl('getPlanlar')); }
      } else { data = await jsonpFetch(buildUrl('getPlanlar')); }
    } catch(e) { data = await jsonpFetch(buildUrl('getPlanlar')); }
    if (data && data.status === 'OK' && Array.isArray(data.planlar)) {
      // Sheets verisini local ile birleştir (id bazlı tekrar önleme)
      const mevcutIdler = new Set(PLAN_OKULLAR.map(p => p.id));
      const yeniler = data.planlar.filter(p => !mevcutIdler.has(p.id));
      PLAN_OKULLAR = [...PLAN_OKULLAR, ...yeniler];
      try { localStorage.setItem('ISG_PLAN_OKULLAR', JSON.stringify(PLAN_OKULLAR)); } catch(e){}
      planListeRender();
    }
  } catch(e) { console.warn('Plan yükleme hatası:', e); }
}

globalThis.planListeRender = function() {
  const liste     = document.getElementById('planListe');
  const sayacEl   = document.getElementById('planListeSayac');
  const toggleBtn = document.getElementById('planListeToggleBtn');
  if (!liste) return;
  const toplam = PLAN_OKULLAR.length;
  if (sayacEl) sayacEl.textContent = toplam + ' okul planlandı';
  if (toggleBtn) toggleBtn.style.display = toplam > 0 ? 'flex' : 'none';
  if (toplam === 0) {
    liste.innerHTML = '<div class="plan-bos">Henüz planlanan denetim yok</div>';
    return;
  }
  // Tamamlanan okul adlarını bul (DENETIMLER ile eşleştir)
  const tamamlananlar = new Set(DENETIMLER.map(d => (d.okul||'').toLowerCase().trim()));
  const sorted = [...PLAN_OKULLAR].sort((a,b) => (a.tarih||'').localeCompare(b.tarih||''));
  liste.innerHTML = sorted.map(p => {
    const tamamlandi = tamamlananlar.has((p.okul||'').toLowerCase().trim());
    const tarihGoster = p.tarih ? p.tarih.split('-').reverse().join('.') : '\u2013';
    const detaylar = [p.tur, p.mudur, p.tel, p.adres].filter(Boolean).join(' \u00b7 ');
    const detayHtml = detaylar ? '<div class="plan-okul-detay">'+detaylar+'</div>' : '';
    const durumCls  = tamamlandi ? 'pod-tamamlandi' : 'pod-bekliyor';
    const durumIco  = tamamlandi ? '\u2713' : '\u23f3';
    return '<div class="plan-okul-row">'
      + '<div class="plan-okul-durum '+durumCls+'">'+durumIco+'</div>'
      + '<div class="plan-okul-bilgi">'
      + '<div class="plan-okul-ad" title="'+p.okul+'">'+p.okul+'</div>'
      + detayHtml
      + '</div>'
      + '<div class="plan-okul-tarih">'+tarihGoster+'</div>'
      + '<button class="plan-okul-sil" onclick="planOkulSil('+p.id+')" title="Sil">\u2715</button>'
      + '</div>';
  }).join('');
}

globalThis.ozetAnalizRender = function(v) {
  planYukle();
  planListeRender();
  if (APPS_SCRIPT_URL) planSheetstenYukle();

  const simdi    = new Date();
  const buYil    = simdi.getFullYear();
  const buAy     = simdi.getMonth(); // 0-11
  const gecenGun = Math.floor((simdi - new Date(buYil, 0, 1)) / 86400000) + 1;
  const yilGun   = (buYil % 4 === 0) ? 366 : 365;

  // ── 1. PLAN vs GERÇEKLEŞME ──
  const topDenetim = v.length;
  const hedef      = PLAN_VERI.hedef || 1;
  const pct        = Math.min(100, Math.round(topDenetim / hedef * 100));
  const kalan      = Math.max(0, hedef - topDenetim);

  const fillEl = document.getElementById('planFill');
  const pctEl  = document.getElementById('planPct');
  if (fillEl) fillEl.style.width = pct + '%';
  if (pctEl)  pctEl.textContent  = '%' + pct;

  // Renk: hedefe göre beklenen ilerlemeyle karşılaştır
  const beklenenPct = Math.round(gecenGun / yilGun * 100);
  const fark        = pct - beklenenPct;
  const renkSinif   = fark >= 0 ? 'green' : fark >= -10 ? 'amber' : 'red';
  if (pctEl) { pctEl.className = 'plan-prog-pct ' + renkSinif; }

  const yapEl  = document.getElementById('planYapilan');
  const kalEl  = document.getElementById('planKalan');
  if (yapEl) yapEl.textContent = topDenetim + ' denetim yapıldı  (beklenen: ' + Math.round(hedef * gecenGun / yilGun) + ')';
  if (kalEl) kalEl.textContent = kalan + ' kalan';

  // ── 2. AYLIK TREND ──
  const aylik = Array(12).fill(0);
  v.forEach(d => {
    if (!d.tarih) return;
    // tarih formatı: GG.AA.YYYY veya YYYY-MM-DD
    let ay = -1;
    if (d.tarih.includes('.')) {
      const parts = d.tarih.split('.');
      if (parts.length >= 2) ay = parseInt(parts[1]) - 1;
    } else if (d.tarih.includes('-')) {
      const parts = d.tarih.split('-');
      if (parts.length >= 2) ay = parseInt(parts[1]) - 1;
    }
    if (ay >= 0 && ay < 12) aylik[ay]++;
  });

  const ayHedef   = Math.round(hedef / 12);
  const ayMaxVal  = Math.max(...aylik, ayHedef, 1);
  const gridEl    = document.getElementById('ayGrid');
  if (gridEl) {
    gridEl.innerHTML = AYLAR.map((ad, i) => {
      const sayi    = aylik[i];
      const barPct  = Math.round(sayi / ayMaxVal * 100);
      const renk    = i > buAy ? 'rgba(100,116,139,.3)'
                    : sayi >= ayHedef ? '#34d399'
                    : sayi > 0 ? '#fbbf24' : '#f87171';
      const aktif   = i === buAy ? 'border:1px solid rgba(56,189,248,.4);border-radius:4px;' : '';
      return '<div class="ay-col" style="' + (aktif) + '">\n        <div class="ay-bar-wrap">\n          <div class="ay-bar-gercek" style="height:' + (barPct) + '%;background:' + (renk) + '"></div>\n        </div>\n        <div class="ay-lbl">' + (ad) + '</div>\n        <div class="ay-sayi" style="color:' + (renk) + '">' + (sayi||'') + '</div>\n      </div>';
    }).join('');
  }

  // ── 3. KAPSAM ──
  const okulSayac  = {};
  v.forEach(d => { okulSayac[d.okul] = (okulSayac[d.okul] || 0) + 1; });
  const benzersiz  = Object.keys(okulSayac).length;
  const coklu      = Object.values(okulSayac).filter(c => c > 1).length;
  const toplamOkul = PLAN_VERI.toplam || 1;
  const kapsanmadi = Math.max(0, toplamOkul - benzersiz);
  const kapsamPct  = Math.min(100, Math.round(benzersiz / toplamOkul * 100));

  // Donut güncelle (çevre: 2*pi*20 ≈ 125.6)
  const donutEl = document.getElementById('kapsamDonut');
  if (donutEl) {
    const offset = 125.6 - (125.6 * kapsamPct / 100);
    donutEl.setAttribute('stroke-dashoffset', offset.toFixed(1));
    donutEl.setAttribute('stroke', kapsamPct >= 70 ? '#34d399' : kapsamPct >= 40 ? '#fbbf24' : '#f87171');
  }
  const kpEl = document.getElementById('kapsamPct');
  if (kpEl) {
    kpEl.textContent = '%' + kapsamPct;
    kpEl.className = 'kapsam-donut-pct ' + (kapsamPct >= 70 ? 'green' : kapsamPct >= 40 ? 'amber' : 'red');
  }

  const set = (id, val) => { const el = document.getElementById(id); if(el) el.textContent = val; };
  set('kapsamBenzersiz', benzersiz + ' okul');
  set('kapsamToplam',    topDenetim + ' denetim');
  set('kapsamKalan',     kapsanmadi + ' okul');
  set('kapsamCoklu',     coklu + ' okul');

  // ── 4. SİSTEMATİK SORUNLAR ──
  const maddeHata = {};
  v.forEach(d => {
    if (!d.answers) return;
    Object.entries(d.answers).forEach(([no, val]) => {
      if (val === 'Hayır/Yok' || val === 'Eksik') {
        const n = parseInt(no);
        maddeHata[n] = (maddeHata[n] || 0) + 1;
      }
    });
  });

  const topMaddeler = Object.entries(maddeHata)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 5);

  const maxHata = topMaddeler[0]?.[1] || 1;
  const sisEl   = document.getElementById('sisProblems');
  if (sisEl) {
    sisEl.innerHTML = topMaddeler.length > 0
      ? topMaddeler.map(([no, cnt]) => {
          const ad    = MADDE_ADLARI_DASH[no] || ('Madde ' + no);
          const adKisa = ad.length > 45 ? ad.substring(0, 45) + '…' : ad;
          const barPct = Math.round(cnt / maxHata * 100);
          return '<div class="sis-row">\n            <div class="sis-no">' + (no) + '.</div>\n            <div class="sis-ad">' + (adKisa) + '</div>\n            <div class="sis-bar-wrap"><div class="sis-bar-fill" style="width:' + (barPct) + '%"></div></div>\n            <div class="sis-cnt">' + (cnt) + '</div>\n          </div>';
        }).join('')
      : '<div style="font-size:11px;color:var(--mut);padding:8px 0">Madde bazlı veri için form üzerinden denetim girin</div>';
  }
}
globalThis.toast = function(msg, cls = '') {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.className   = 'toast show ' + cls;
  clearTimeout(t._timer);
  t._timer = setTimeout(() => t.classList.remove('show'), 3200);
}

let tabloTumVeri  = [];
let tabloModTum   = false;  // false=özet(10), true=tümü(sayfalama)
let tabloSayfa    = 0;
const TABLO_SAYFA_BOYUT = 20;

globalThis.tabloRender = function() {
  const arama   = (document.getElementById('tabloArama')?.value || '').toLowerCase().trim();
  const kontrol = document.getElementById('tabloKontroller');
  const ozetEl  = document.getElementById('tabloOzetYazi');
  const tumBtn  = document.getElementById('tumunuGoster');

  // Arama filtresi
  const filtrelenmis = arama
    ? tabloTumVeri.filter(d => (d.okul||'').toLowerCase().includes(arama))
    : tabloTumVeri;

  let gosterilecek;

  if (!tabloModTum) {
    // ÖZET MOD: son 10
    gosterilecek = filtrelenmis.slice(0, 10);
    if (kontrol) kontrol.style.display = 'none';
    if (ozetEl)  ozetEl.textContent = 'Son 10 denetim gösteriliyor (toplam ' + (filtrelenmis.length) + ')';
    if (tumBtn)  tumBtn.style.display = filtrelenmis.length > 10 ? 'block' : 'none';
    document.getElementById('tabloSayfalama').innerHTML = '';
  } else {
    // TAM MOD: sayfalama
    const toplamSayfa = Math.ceil(filtrelenmis.length / TABLO_SAYFA_BOYUT);
    if (tabloSayfa >= toplamSayfa) tabloSayfa = Math.max(0, toplamSayfa - 1);
    const bas = tabloSayfa * TABLO_SAYFA_BOYUT;
    gosterilecek = filtrelenmis.slice(bas, bas + TABLO_SAYFA_BOYUT);
    if (kontrol) kontrol.style.display = 'flex';
    if (tumBtn)  tumBtn.style.display  = 'none';
    const gosterSayac = document.getElementById('tabloSayac');
    if (gosterSayac) gosterSayac.textContent =
      (bas+1) + '–' + (Math.min(bas+TABLO_SAYFA_BOYUT, filtrelenmis.length)) + ' / ' + (filtrelenmis.length);
    if (ozetEl) ozetEl.textContent = '';
    // Sayfa butonları
    _tabloSayfaButonlari(toplamSayfa);
  }

  // Satırları oluştur
  document.getElementById('tabloBody').innerHTML = gosterilecek.map(d => {
    const uyg       = (d.hayir||0) + (d.eksik||0);
    const pct       = (uyg / 138 * 100).toFixed(1);
    const raporLink = d.raporUrl ? '<a class="tbl-link" href="' + (d.raporUrl) + '" target="_blank">📄 Doc</a>' : '';
    const pdfLink   = d.pdfUrl   ? '<a class="tbl-link" href="' + (d.pdfUrl) + '"   target="_blank">📕 PDF</a>' : '';
    const linkler   = raporLink||pdfLink ? raporLink+' '+pdfLink : '<span style="color:var(--mut);font-size:10px">–</span>';
    const ebysBadge = (d.ebys||'').includes('Kapatıldı')
      ? '<span class="badge b-kap">✅ Kapatıldı</span>'
      : '<span class="badge b-bek">'+(d.ebys||'⏳ Bekliyor')+'</span>';
    return '<tr>\n      <td>' + (d.okul.length>36?d.okul.substring(0,36)+'…':d.okul) + '</td>\n      <td style="color:' + (TUR_RENK[d.tur]||'#64748b') + ';font-weight:600">' + (d.tur||'–') + '</td>\n      <td>' + (d.tarih||'–') + '</td>\n      <td><span class="badge ' + (d.tehlike==='Tehlikeli'||d.tehlike==='Çok Tehlikeli'?'b-teh':'b-az') + '">' + (d.tehlike||'–') + '</span></td>\n      <td style="font-family:Arial Black,sans-serif;font-weight:700;color:' + (uyg>=30?'#f87171':uyg>=10?'#fbbf24':'#34d399') + '">' + (uyg) + ' <span style="font-size:10px;color:var(--mut)">(%' + (pct) + ')</span></td>\n      <td>' + (ebysBadge) + '</td>\n      <td>' + (linkler) + '</td>\n    </tr>';
  }).join('') || '<tr><td colspan="7" style="text-align:center;color:var(--mut);padding:18px">' + (arama ? 'Arama sonucu bulunamadı' : 'Veri yok') + '</td></tr>';
}

globalThis._tabloSayfaButonlari = function(toplamSayfa) {
  const el = document.getElementById('tabloSayfalama');
  if (!el || toplamSayfa <= 1) { if(el) el.innerHTML=''; return; }
  const stil = (aktif) =>
    'style="padding:4px 10px;border-radius:7px;border:1px solid ' + (aktif?'rgba(56,189,248,.5)':'var(--bdr)') + ';\n     background:' + (aktif?'rgba(56,189,248,.15)':'transparent') + ';color:' + (aktif?'var(--teal)':'var(--mut2)') + ';\n     font-size:12px;font-weight:700;cursor:pointer;font-family:Arial,sans-serif;"';

  let html = '<button onclick="tabloSayfaDegis(-1)" ' + (stil(false)) + '>←</button>';
  for (let i = 0; i < toplamSayfa; i++) {
    if (toplamSayfa > 7 && Math.abs(i - tabloSayfa) > 2 && i !== 0 && i !== toplamSayfa-1) {
      if (i === 1 || i === toplamSayfa-2) html += '<span style="color:var(--mut);padding:0 4px">…</span>';
      continue;
    }
    html += '<button onclick="tabloSayfaDegis(' + (i) + ',true)" ' + (stil(i===tabloSayfa)) + '>' + (i+1) + '</button>';
  }
  html += '<button onclick="tabloSayfaDegis(1)" ' + (stil(false)) + '>→</button>';
  el.innerHTML = html;
}

globalThis.tabloSayfaDegis = function(val, mutlak = false) {
  const arama = (document.getElementById('tabloArama')?.value||'').toLowerCase().trim();
  const filtre = arama ? tabloTumVeri.filter(d=>(d.okul||'').toLowerCase().includes(arama)) : tabloTumVeri;
  const toplamSayfa = Math.ceil(filtre.length / TABLO_SAYFA_BOYUT);
  tabloSayfa = mutlak ? val : Math.max(0, Math.min(toplamSayfa-1, tabloSayfa + val));
  tabloRender();
}

}
